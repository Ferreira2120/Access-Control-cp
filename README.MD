import { createBrowserRouter, RouterProvider } from "react-router-dom";
import App from "./App";
import Login from "./routes/Login";
import Cadastro from "./routes/Cadastro";

{ path: "/", element: <App /> },

<StrictMode>
<RouterProvider router={router} />
